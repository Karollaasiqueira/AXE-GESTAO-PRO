echo const jwt = require('jsonwebtoken'^);
echo.
echo // Middleware para verificar token JWT
echo exports.authenticate = (req, res, next^) =^> {
echo   try {
echo     const authHeader = req.headers.authorization;
echo.
echo     if (!authHeader ^|^| !authHeader.startsWith('Bearer '^)^) {
echo       return res.status(401^).json({ error: 'Token não fornecido' }^);
echo     }
echo.
echo     const token = authHeader.substring(7^);
echo     const decoded = jwt.verify(token, process.env.JWT_SECRET^);
echo.
echo     req.userId = decoded.userId;
echo     req.userRole = decoded.role;
echo.
echo     next(^);
echo.
echo   } catch (error^) {
echo     return res.status(401^).json({ error: 'Token inválido ou expirado' }^);
echo   }
echo };
echo.
echo // Middleware para verificar permissões
echo exports.authorize = (...roles^) =^> {
echo   return (req, res, next^) =^> {
echo     if (!roles.includes(req.userRole^)^) {
echo       return res.status(403^).json({ error: 'Acesso negado' }^);
echo     }
echo     next(^);
echo   };
echo };
